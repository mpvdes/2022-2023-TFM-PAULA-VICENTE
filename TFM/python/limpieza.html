pib = pd.read_excel('PIB_2021.xlsx')
def quitar_espacios_importes(x):
    l = x.split()
    d = ''.join(x for x in l)    
    return int(d)


def quitar_espacios_nombres_pais(x):
    l = x.strip()    
    return l


pib['PIB (float)'] = pib['PIB (nominal millones de USD)'].map(quitar_espacios_importes)
